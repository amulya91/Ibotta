CREATE TABLE offer_rewards (
    ID INTEGER,
    OFFER_ID INTEGER,
    TYPE TEXT,
    AMOUNT NUMERIC,
    CREATED_AT TIMESTAMP,
    UPDATED_AT TIMESTAMP
);


CREATE TABLE customer_offers (
    ID INTEGER,
    CUSTOMER_ID INTEGER,
    OFFER_ID INTEGER,
    ACTIVATED TIMESTAMP,
    VERIFIED TIMESTAMP
);

CREATE TABLE customer_offer_rewards (
    ID INTEGER,
    CUSTOMER_ID INTEGER,
    OFFER_REWARD_ID INTEGER,
    FINISHED TIMESTAMP,
    CREATED_AT TIMESTAMP
);


CREATE TABLE customer_offer_redemptions (
    ID INTEGER,
    CUSTOMER_OFFER_ID INTEGER,
    VERIFIED_REDEMPTION_COUNT INTEGER,
    SUBMITTED_REDEMPTION_COUNT INTEGER,
    OFFER_AMOUNT NUMERIC,
    CREATED_AT TIMESTAMP
);
